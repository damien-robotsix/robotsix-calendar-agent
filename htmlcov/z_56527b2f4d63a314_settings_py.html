1"""Single source of truth for all environment-variable configuration. 

2 

3Uses :class:`pydantic_settings.BaseSettings` to read, validate, and 

4normalise configuration from the process environment. 

5""" 

6 

7from __future__ import annotations 

8 

9import logging 

10from typing import Any 

11 

12from pydantic import SecretStr, field_validator 

13from pydantic_settings import BaseSettings, SettingsConfigDict 

14 

15 

16class Settings(BaseSettings): 

17 """Application settings loaded from environment variables. 

18 

19 All RADICALE_* and BROKER_HOST/BROKER_AGENT_TOKEN fields have empty 

20 defaults so that the existing constructor-argument fallback and 

21 emptiness checks in ``CalendarAgent.__init__`` and 

22 ``_build_brokered_agent`` continue to work unchanged. 

23 """ 

24 

25 model_config = SettingsConfigDict(extra="ignore") 

26 

27 # -- Radicale credentials ------------------------------------------------ 

28 RADICALE_URL: str = "" 

29 RADICALE_USERNAME: str = "" 

30 RADICALE_PASSWORD: SecretStr = SecretStr("") 

31 RADICALE_DEFAULT_CALENDAR: str = "Robotsix" 

32 

33 # -- Calendar agent identity / transport --------------------------------- 

34 CALENDAR_AGENT_TRANSPORT: str = "inprocess" 

35 CALENDAR_AGENT_ID: str = "robotsix-calendar" 

36 

37 # -- Broker connection --------------------------------------------------- 

38 BROKER_HOST: str = "" 

39 BROKER_PORT: int = 9090 

40 BROKER_SCHEME: str = "https" 

41 BROKER_AGENT_TOKEN: SecretStr = SecretStr("") 

42 BROKER_TLS_CA: str | None = None 

43 BROKER_CLIENT_CERT: str | None = None 

44 BROKER_CLIENT_KEY: str | None = None 

45 

46 # -- Logging ------------------------------------------------------------- 

47 LOG_LEVEL: str = "INFO" 

48 JSON_LOGS: bool = False

50 # -- Validators ---------------------------------------------------------- 

51 

52 @field_validator("CALENDAR_AGENT_TRANSPORT") 

53 @classmethod 

54 def _normalize_transport(cls, v: str) -> str: 

55 """Strip and lower-case the transport mode name. 

56 

57 The caller (``main()``) is still responsible for rejecting 

58 unrecognised values so that the existing ``ValueError`` 

59 message — which includes the offending value — is preserved. 

60 """ 

61 return v.strip().lower() 

62 

63 @field_validator("BROKER_PORT", mode="before") 

64 @classmethod 

65 def _validate_port(cls, v: Any) -> int: 

66 """Validate that *v* is an integer in the range 1-65535. 

67 

68 Replaces the bare ``int(os.environ.get("BROKER_PORT", …))`` cast 

69 with a clear, field-specific message on failure. 

70 """ 

71 _PORT_MSG = "BROKER_PORT must be an integer between 1 and 65535" 

72 if isinstance(v, int): 

73 port = v 

74 else: 

75 try: 

76 port = int(v) 

77 except (TypeError, ValueError): 

78 raise ValueError(_PORT_MSG) from None 

79 if port < 1 or port > 65535: 

80 raise ValueError(_PORT_MSG) 

81 return port 

82 

83 @field_validator( 

84 "BROKER_TLS_CA", 

85 "BROKER_CLIENT_CERT", 

86 "BROKER_CLIENT_KEY", 

87 mode="before", 

88 ) 

89 @classmethod 

90 def _empty_str_to_none(cls, v: Any) -> Any: 

91 """Treat empty strings as ``None``. 

92 

93 Mirrors the ``or None`` idiom previously used in 

94 ``_build_brokered_agent`` for the mTLS-related variables. 

95 """ 

96 if isinstance(v, str) and v == "": 

97 return None 

98 return v 

99 

100 @field_validator("LOG_LEVEL") 

101 @classmethod 

102 def _normalize_log_level(cls, v: str) -> str: 

103 """Normalise to uppercase and reject invalid log levels.""" 

104 v = v.strip().upper() 

105 if v not in logging.getLevelNamesMapping(): 

106 raise ValueError( 

107 f"Invalid LOG_LEVEL={v!r}; must be one of " 

108 f"{sorted(logging.getLevelNamesMapping())}" 

109 ) 

110 return v
